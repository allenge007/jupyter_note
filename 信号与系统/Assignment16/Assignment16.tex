  第一项\\
  \[
  L\{e^{-4t}u(t)\} = \frac{1}{s+4},\quad \text{ROC: } \Re(s) > -4.
  \]
\item
  第二项 \[
  L\{e^{-at}\sin(bt)u(t)\} = \frac{b}{(s+a)^2+b^2},\quad \text{ROC: } \Re(s) > -a,
  \]\\
  其中 \(a=5\), \(b=5\) \[
  L\{e^{-5t}\sin(5t)u(t)\} = \frac{5}{(s+5)^2+25},\quad \text{ROC: } \Re(s) > -5.
  \]
\end{itemize}

因此，整体拉普拉斯变换为\\
\[
X(s) = \frac{1}{s+4} + \frac{5}{(s+5)^2+25},
\]\\
其 ROC 为两部分 ROC 的交集，即\\
\[
\Re(s) > -4.
\]

\begin{itemize}
\tightlist
\item
  \textbf{极点}

  \begin{itemize}
  \tightlist
  \item
    第一项在 \(s=-4\) 有一个极点。\\
  \item
    第二项中解方程\\
    \[
    (s+5)^2+25=0 \quad\Rightarrow\quad s+5=\pm j5 \quad\Rightarrow\quad s=-5\pm j5.
    \]\\
    因此，另外得到一对复共轭极点：\(s=-5\pm j5\)。
  \end{itemize}
\item
  \textbf{零点}\\
  若对两个分式统一通分后得到合并表达式，分子为\\
  \[
  N(s) = (s+5)^2+25+5(s+4),
  \]\\
  展开计算可得\\
  \[
  N(s) = s^2+15s+70.
  \]\\
  零点满足\\
  \[
  s^2+15s+70 = 0.
  \]\\
  因为判别式\\
  \[
  \Delta=15^2-4\cdot70=225-280=-55,
  \]\\
  得到一对复零点：\\
  \[
  s=-\frac{15}{2}\pm j\frac{\sqrt{55}}{2}.
  \]
\end{itemize}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsubsection{(j)}\label{j}

\begin{itemize}
\item
  对于 \(\delta(3t)\)，利用冲激函数的缩放性质\\
  \[
  \delta(3t)=\frac{1}{3}\delta(t) \quad\Rightarrow\quad L\{\delta(3t)\}=\frac{1}{3}.
  \]
\item
  对于 \(u(3t)\)，由于单位阶跃函数在 \(t\ge0\) 下恒成立\\
  \[
  u(3t)=u(t) \quad\Rightarrow\quad L\{u(3t)\}=L\{u(t)\}=\frac{1}{s},\quad \text{ROC: } \Re(s)>0.
  \]
\end{itemize}

因此，拉普拉斯变换为\\
\[
X(s)=\frac{1}{3}+\frac{1}{s}=\frac{s+3}{3s},
\]\\
其 ROC 为\\
\[
\Re(s)>0.
\]

\begin{itemize}
\item
  \textbf{极点}\\
  分式 \(\frac{1}{s}\) 提供一个极点在 \(s=0\)。
\item
  \textbf{零点}\\
  分子 \(s+3=0\) 给出一个零点在 \(s=-3\)。
\end{itemize}
